    refreshIfVolumeTopN(LocalDate today):
    ```
    for rule in ruleRepo.findAll() where RuleStatus.isLiveActive(rule):
        def = parse(rule.definition); u = def.universe()
        if u == null || !"volume_top_n".equals(u.type()): continue   // Pitfall 3
        market = u.market() != null ? u.market() : "KOSPI"
        topN   = u.topN() != null ? u.topN() : 10
        LinkedHashSet<String> merged = new LinkedHashSet<>(liveRanking.topVolume(market, today, topN, true))
        if u.additionalSymbols() != null: merged.addAll(u.additionalSymbols())
        // 보유 포지션 union (Pitfall 2 — 이탈 종목 청산 위해 수집 대상 유지)
        account = accountRepo.findByUserId(rule.getUserId()).orElse(null)
        if account != null:
            for pos in positionRepo.findByAccountId(account.getId()): merged.add(pos.getSymbol())
        paperLiveSymbolService.assignSymbols(rule.getId(), merged)
    ```
